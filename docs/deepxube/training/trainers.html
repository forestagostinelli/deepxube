 30class Status:
 31    def __init__(self, step_max: int, balance_steps: bool):
 32        self.itr: int = 0
 33        self.update_num: int = 0
 34        self.targ_update_num: int = 0
 35        self.step_max: int = step_max
 36        self.step_probs: NDArray
 37        self.step_max_curr: int = 1
 38        if balance_steps:
 39            self.step_probs = np.zeros(self.step_max + 1)
 40            self.step_probs[0:2] = 0.5
 41            self.step_probs[2:] = 0
 42        else:
 43            self.step_probs = np.ones(self.step_max + 1)/(self.step_max + 1)
 44        self.per_solved_best: float = 0.0

 46    def update_step_probs(self, step_to_search_perf: Dict[int, PathFindPerf]) -> None:
 47        ave_solve: float = float(np.mean([step_to_search_perf[step].per_solved()
 48                                          for step in step_to_search_perf.keys()]))
 49        if ave_solve >= 50.0:
 50            self.step_max_curr = min(self.step_max_curr * 2, self.step_max)
 51
 52        self.step_probs = np.zeros(self.step_max + 1)
 53        self.step_probs[np.arange(0, self.step_max_curr + 1)] = 1 / (self.step_max_curr + 1)

111    def update_step(self) -> None:
112        self.db.clear()
113
114        # print info
115        start_info_l: List[str] = [f"itr: {self.status.itr}", f"update_num: {self.status.update_num}",
116                                   f"targ_update: {self.status.targ_update_num}"]
117
118        num_gen: int = self.train_args.batch_size * self.updater.up_args.get_up_gen_itrs()
119        start_info_l.append(f"num_gen: {format(num_gen, ',')}")
120        if self.train_args.balance_steps:
121            start_info_l.append(f"step max (curr): {self.status.step_max_curr}")
122        print(f"\nGetting Data - {', '.join(start_info_l)}")
123        times: Times = Times()
124
125        # start updater
126        start_time = time.time()
127        self.updater.start_update(self.status.step_probs.tolist(), num_gen, self.status.targ_update_num,
128                                  self.train_args.batch_size, self.device, self.on_gpu)
129        times.record_time("up_start", time.time() - start_time)
130
131        # do training
132        loss: float
133        ctgs_l: List[NDArray]
134        if not self.updater.up_args.sync_main:
135            ctgs_l = self._get_update_data(num_gen, times)
136            self._end_update(ctgs_l, times)
137            loss = self._train_no_sync_main(times)
138        else:
139            loss, ctgs_l = self._train_sync_main(num_gen, times)
140            self._end_update(ctgs_l, times)
141
142        # save nnet
143        start_time = time.time()
144        torch.save(self.nnet.state_dict(), self.nnet_file)
145        times.record_time("save", time.time() - start_time)
146
147        # update nnet
148        update_targ: bool = False
149        if loss < self.train_args.loss_thresh:
150            if self.train_args.targ_up_searches <= 0:
151                update_targ = True
152            else:
153                start_time = time.time()
154                per_solved: float = self._update_greedy_perf(self.status.targ_update_num + 1)
155                print(f"Greedy policy solved (best): {per_solved:.2f}% ({self.status.per_solved_best:.2f}%)")
156                if per_solved > self.status.per_solved_best:
157                    update_targ = True
158                    self.status.per_solved_best = per_solved
159                times.record_time("greedy_policy_test", time.time() - start_time)
160
161        if update_targ:
162            shutil.copy(self.nnet_file, self.nnet_targ_file)
163            self.status.targ_update_num = self.status.targ_update_num + 1
164        self.status.update_num += 1
165
166        # noinspection PyTypeChecker
167        pickle.dump(self.status, open(self.status_file, "wb"), protocol=-1)
168        print(f"Train - itrs: {self.updater.up_args.up_itrs}, loss: {loss:.2E}, targ_updated: {update_targ}")
169        print(f"Times - {times.get_time_str()}")

171    def _get_update_data(self, num_gen: int, times: Times) -> List[NDArray]:
172        start_time = time.time()
173        ctgs_l: List[NDArray] = []
174        while self.db.size() < num_gen:
175            data_l: List[List[NDArray]] = self.updater.get_update_data()
176            for data in data_l:
177                ctgs_l.append(data[-1])
178                self.db.add(data)
179        times.record_time("up_data", time.time() - start_time)
180
181        return ctgs_l

183    def _train_no_sync_main(self, times: Times) -> float:
184        # train
185        loss: float = np.inf
186        for _ in range(self.updater.up_args.up_itrs):
187            # sample data
188            start_time = time.time()
189            batch: List[NDArray] = self.db.sample(self.train_args.batch_size)
190            times.record_time("data_samp", time.time() - start_time)
191
192            # train
193            loss = self._train_itr(batch[:-1], batch[-1], times)
194
195        return loss

197    def _train_sync_main(self, num_gen: int, times: Times) -> Tuple[float, List[NDArray]]:
198        loss: float = np.inf
199        ctgs_l: List[NDArray] = []
200        update_train_itr: int = 0
201        while update_train_itr < self.updater.up_args.up_itrs:
202            batch: List[NDArray]
203            # data from updater should not be more that train_args.batch_size
204            start_time = time.time()
205            if self.db.size() == num_gen:
206                batch = self.db.sample(self.train_args.batch_size)
207            else:
208                self.nnet.eval()
209                while self.db.size() < ((update_train_itr + 1) * self.train_args.batch_size):
210                    # get heuristic values for ongoing search
211                    q_res: Optional[Tuple[int, List[SharedNDArray]]] = get_nowait_noerr(self.to_main_q)
212                    if q_res is not None:
213                        proc_id, inputs_np_shm = q_res
214                        nnet_in_out_shared_q(self.nnet, inputs_np_shm, self.updater.up_args.nnet_batch_size,
215                                             self.device, self.from_main_qs[proc_id])
216
217                    # get update data
218                    data_l_i: List[List[NDArray]] = self.updater.get_update_data(nowait=True)
219                    for data in data_l_i:
220                        ctgs_l.append(data[-1])
221                        self.db.add(data)
222                sel_idxs: NDArray = np.arange(update_train_itr * self.train_args.batch_size,
223                                              (update_train_itr + 1) * self.train_args.batch_size)
224                batch = sel_l(self.db.arrays, sel_idxs)
225
226            times.record_time("up_data", time.time() - start_time)
227
228            # train
229            start_time = time.time()
230            loss = self._train_itr(batch[:-1], batch[-1], times)
231            update_train_itr += 1
232            times.record_time("train", time.time() - start_time)
233
234        return loss, ctgs_l

236    def _train_itr(self, inputs_batch_np: List[NDArray], ctgs_batch_np: NDArray, times: Times) -> float:
237        start_time = time.time()
238        ctgs_batch_np = np.expand_dims(ctgs_batch_np.astype(np.float32), 1)
239        self.nnet.train()
240        loss: float = train_heur_nnet_step(self.nnet, inputs_batch_np, ctgs_batch_np, self.optimizer, self.criterion,
241                                           self.device, self.status.itr, self.train_args)
242        self.status.itr += 1
243        times.record_time("train", time.time() - start_time)
244        return loss

246    def _end_update(self, ctgs_l: List[NDArray], times: Times) -> None:
247        start_time = time.time()
248        step_to_search_perf: Dict[int, PathFindPerf] = self.updater.end_update()
249        if self.train_args.balance_steps:
250            self.status.update_step_probs(step_to_search_perf)
251
252        per_solved_ave, path_costs_ave, search_itrs_ave = get_eq_weighted_perf(step_to_search_perf)
253        ctgs_mean, ctgs_min, ctgs_max = ctgs_summary(ctgs_l)
254
255        self.writer.add_scalar("train/pathfind/solved", per_solved_ave, self.status.itr)
256        self.writer.add_scalar("train/pathfind/path_cost", path_costs_ave, self.status.itr)
257        self.writer.add_scalar("train/pathfind/search_itrs", search_itrs_ave, self.status.itr)
258
259        self.writer.add_scalar("train/ctgs/mean", ctgs_mean, self.status.itr)
260        self.writer.add_scalar("train/ctgs/min", ctgs_min, self.status.itr)
261        self.writer.add_scalar("train/ctgs/max", ctgs_max, self.status.itr)
262
263        post_up_info_l: List[str] = [f"%solved: {per_solved_ave:.2f}", f"path_costs: {path_costs_ave:.3f}",
264                                     f"search_itrs: {search_itrs_ave:.3f}",
265                                     f"cost-to-go (mean/min/max): {ctgs_mean:.2f}/{ctgs_min:.2f}/{ctgs_max:.2f}"]
266
267        print(f"Data - {', '.join(post_up_info_l)}")
268        times.record_time("up_end", time.time() - start_time)

270    def _update_greedy_perf(self, update_num: int) -> float:
271        # get updater
272        updater_greedy: UpdateHeur
273        heur_nnet: HeurNNetPar = self.updater.get_heur_nnet()
274        up_greedy_args: UpGreedyPolicyArgs = UpGreedyPolicyArgs(0.0, 0.0)
275        up_heur_args: UpHeurArgs = UpHeurArgs(False, 1)
276        up_args = dataclasses.replace(self.updater.up_args)
277        up_args.sync_main = False
278        if isinstance(heur_nnet, HeurNNetParV):
279            updater_greedy = UpdateHeurGrPolVEnum(self.updater.domain, up_args, up_heur_args, up_greedy_args, heur_nnet)
280        elif isinstance(heur_nnet, HeurNNetParQ):
281            updater_greedy = UpdateHeurGrPolQEnum(self.updater.domain, up_args, up_heur_args, up_greedy_args, heur_nnet)
282        else:
283            raise ValueError(f"Unknown heuristic function type {heur_nnet}")
284
285        # do greedy update
286        updater_greedy.set_heur_file(self.nnet_file)
287        step_probs = np.ones(self.updater.up_args.step_max + 1) / (self.updater.up_args.step_max + 1)
288        num_gen: int = self.updater.up_args.search_itrs * self.train_args.targ_up_searches
289        updater_greedy.start_procs()
290        updater_greedy.start_update(step_probs.tolist(), num_gen, update_num, self.train_args.batch_size,
291                                    self.device, self.on_gpu)
292        while updater_greedy.num_generated < num_gen:
293            self.updater.get_update_data()
294
295        step_to_search_perf: Dict[int, PathFindPerf] = updater_greedy.end_update()
296        updater_greedy.stop_procs()
297        per_solved_ave, path_cost_ave, search_itrs_ave = get_eq_weighted_perf(step_to_search_perf)
298        # print(f"%solved: {per_solved_ave:.2f}, path_costs: {path_cost_ave:.3f}, search_itrs: {search_itrs_ave:.3f} "
299        #      f"(greedy perf)")
300
301        return per_solved_ave

198    def _train_sync_main(self, num_gen: int, times: Times) -> Tuple[float, List[NDArray]]:
199        loss: float = np.inf
200        ctgs_l: List[NDArray] = []
201        update_train_itr: int = 0
202        while update_train_itr < self.updater.up_args.up_itrs:
203            batch: List[NDArray]
204            # data from updater should not be more that train_args.batch_size
205            start_time = time.time()
206            if self.db.size() == num_gen:
207                batch = self.db.sample(self.train_args.batch_size)
208            else:
209                self.nnet.eval()
210                while self.db.size() < ((update_train_itr + 1) * self.train_args.batch_size):
211                    # get heuristic values for ongoing search
212                    q_res: Optional[Tuple[int, List[SharedNDArray]]] = get_nowait_noerr(self.to_main_q)
213                    if q_res is not None:
214                        proc_id, inputs_np_shm = q_res
215                        nnet_in_out_shared_q(self.nnet, inputs_np_shm, self.updater.up_args.nnet_batch_size,
216                                             self.device, self.from_main_qs[proc_id])
217
218                    # get update data
219                    data_l_i: List[List[NDArray]] = self.updater.get_update_data(nowait=True)
220                    for data in data_l_i:
221                        ctgs_l.append(data[-1])
222                        self.db.add(data)
223                sel_idxs: NDArray = np.arange(update_train_itr * self.train_args.batch_size,
224                                              (update_train_itr + 1) * self.train_args.batch_size)
225                batch = sel_l(self.db.arrays, sel_idxs)
226
227            times.record_time("up_data", time.time() - start_time)
228
229            # train
230            start_time = time.time()
231            loss = self._train_itr(batch[:-1], batch[-1], times)
232            update_train_itr += 1
233            times.record_time("train", time.time() - start_time)
234
235        return loss, ctgs_l
