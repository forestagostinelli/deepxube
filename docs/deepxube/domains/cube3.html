 61class Quaternion:
 62    """Quaternion Rotation:
 63
 64    Class to aid in representing 3D rotations via quaternions.
 65    """
 66
 67    @classmethod
 68    def from_v_theta(cls, v, theta) -> 'Quaternion':  # type: ignore
 69        """
 70        Construct quaternions from unit vectors v and rotation angles theta
 71
 72        Parameters
 73        ----------
 74        v : array_like
 75            array of vectors, last dimension 3. Vectors will be normalized.
 76        theta : array_like
 77            array of rotation angles in radians, shape = v.shape[:-1].
 78
 79        Returns
 80        -------
 81        q : quaternion object
 82            quaternion representing the rotations
 83        """
 84        theta = np.asarray(theta)
 85        v = np.asarray(v)
 86        s = np.sin(0.5 * theta)
 87        c = np.cos(0.5 * theta)
 88
 89        v = v * s / np.sqrt(np.sum(v * v, -1))
 90        x_shape = v.shape[:-1] + (4,)
 91
 92        x: NDArray = np.ones(x_shape).reshape(-1, 4)
 93        x[:, 0] = c.ravel()
 94        x[:, 1:] = v.reshape(-1, 3)
 95        x = x.reshape(x_shape)
 96
 97        return cls(x)
 98
 99    def __init__(self, x: NDArray):
100        self.x = np.asarray(x, dtype=float)
101
102    def __repr__(self) -> str:
103        return "Quaternion:\n" + self.x.__repr__()

105    def __mul__(self, other: 'Quaternion') -> 'Quaternion':
106        # multiplication of two quaternions.
107        # we don't implement multiplication by a scalar
108        sxr = self.x.reshape(self.x.shape[:-1] + (4, 1))
109        oxr = other.x.reshape(other.x.shape[:-1] + (1, 4))
110
111        prod = sxr * oxr
112        return_shape = prod.shape[:-1]
113        prod = prod.reshape((-1, 4, 4)).transpose((1, 2, 0))
114
115        ret = np.array([(prod[0, 0] - prod[1, 1]
116                         - prod[2, 2] - prod[3, 3]),
117                        (prod[0, 1] + prod[1, 0]
118                         + prod[2, 3] - prod[3, 2]),
119                        (prod[0, 2] - prod[1, 3]
120                         + prod[2, 0] + prod[3, 1]),
121                        (prod[0, 3] + prod[1, 2]
122                         - prod[2, 1] + prod[3, 0])],
123                       dtype=float,
124                       order='F').T
125        return self.__class__(ret.reshape(return_shape))
126
127    def as_v_theta(self) -> Tuple[NDArray, NDArray]:
128        """Return the v, theta equivalent of the (normalized) quaternion"""
129        x = self.x.reshape((-1, 4)).T
130
131        # compute theta
132        norm = np.sqrt((x ** 2).sum(0))
133        theta = 2 * np.arccos(x[0] / norm)
134
135        # compute the unit vector
136        v = np.array(x[1:], order='F', copy=True)
137        v /= np.sqrt(np.sum(v ** 2, 0))
138
139        # reshape the results
140        v = v.T.reshape(self.x.shape[:-1] + (3,))
141        theta = theta.reshape(self.x.shape[:-1])
142
143        return v, theta

145    def as_rotation_matrix(self) -> NDArray:
146        """Return the rotation matrix of the (normalized) quaternion"""
147        v, theta = self.as_v_theta()
148
149        shape = theta.shape
150        theta = theta.reshape(-1)
151        v = v.reshape(-1, 3).T
152        c = np.cos(theta)
153        s = np.sin(theta)
154
155        mat = np.array([[v[0] * v[0] * (1. - c) + c,
156                         v[0] * v[1] * (1. - c) - v[2] * s,
157                         v[0] * v[2] * (1. - c) + v[1] * s],
158                        [v[1] * v[0] * (1. - c) + v[2] * s,
159                         v[1] * v[1] * (1. - c) + c,
160                         v[1] * v[2] * (1. - c) - v[0] * s],
161                        [v[2] * v[0] * (1. - c) - v[1] * s,
162                         v[2] * v[1] * (1. - c) + v[0] * s,
163                         v[2] * v[2] * (1. - c) + c]],
164                       order='F').T
165        return mat.reshape(shape + (3, 3))
166
167    def rotate(self, points):  # type: ignore
168        rot_mat = self.as_rotation_matrix()
169        return np.dot(points, rot_mat.T)

172def project_points(points, q: Quaternion, view, vertical) -> NDArray:  # type: ignore
173    """Project points using a quaternion q and a view v
174
175    Parameters
176    ----------
177    points : array_like
178        array of last-dimension 3
179    q : utils.viz_utils.Quaternion
180        quaternion representation of the rotation
181    view : array_like
182        length-3 vector giving the point of view
183    vertical : array_like
184        direction of y-axis for view.  An error will be raised if it
185        is parallel to the view.
186
187    Returns
188    -------
189    proj: array_like
190        array of projected points: same shape as points.
191    """
192    if vertical is None:
193        vertical = [0, 1, 0]
194    points = np.asarray(points)
195    view = np.asarray(view)
196
197    xdir = np.cross(vertical, view).astype(float)
198
199    if np.all(xdir == 0):
200        raise ValueError("vertical is parallel to v")
201
202    xdir /= np.sqrt(np.dot(xdir, xdir))
203
204    # get the unit vector corresponing to vertical
205    ydir = np.cross(view, xdir)
206    ydir /= np.sqrt(np.dot(ydir, ydir))
207
208    # normalize the viewer location: this is the z-axis
209    v2 = np.dot(view, view)
210    zdir = view / np.sqrt(v2)
211
212    # rotate the points
213    rot_mat = q.as_rotation_matrix()
214    r_pts = np.dot(points, rot_mat.T)
215
216    # project the points onto the view
217    dpoint = r_pts - view
218    dpoint_view = np.dot(dpoint, view).reshape(dpoint.shape[:-1] + (1,))
219    dproj = -dpoint * v2 / dpoint_view
220
221    trans = list(range(1, dproj.ndim)) + [0]
222    return np.array([np.dot(dproj, xdir),
223                     np.dot(dproj, ydir),
224                     -np.dot(dpoint, zdir)]).transpose(trans)

337    def _initialize_arrays(self) -> None:

339        # base for each one, and then translate & rotate them into position.
340
341        # Define N^2 translations for each face of the cube
342        cubie_width = 2. / self.N
343        translations = np.array([[[-1 + (i + 0.5) * cubie_width,
344                                   -1 + (j + 0.5) * cubie_width, 0]]
345                                 for i in range(self.N)
346                                 for j in range(self.N)])
347
348        # Create arrays for centroids, faces, stickers
349        face_centroids = []
350        faces = []
351        sticker_centroids = []
352        stickers = []
353        colors = []
354
355        factor = np.array([1. / self.N, 1. / self.N, 1])
356
357        for i in range(6):
358            rot_mat = self.rots[i].as_rotation_matrix()
359            faces_t = np.dot(factor * self.base_face
360                             + translations, rot_mat.T)
361            stickers_t = np.dot(factor * self.base_sticker
362                                + translations, rot_mat.T)
363            face_centroids_t = np.dot(self.base_face_centroid
364                                      + translations, rot_mat.T)
365            sticker_centroids_t = np.dot(self.base_sticker_centroid
366                                         + translations, rot_mat.T)
367            # colors_i = i + np.zeros(face_centroids_t.shape[0], dtype=int)
368            colors_i = np.arange(i * face_centroids_t.shape[0], (i + 1) * face_centroids_t.shape[0])
369
370            # append face ID to the face centroids for lex-sorting

340        # base for each one, and then translate & rotate them into position.
341
342        # Define N^2 translations for each face of the cube
343        cubie_width = 2. / self.N
344        translations = np.array([[[-1 + (i + 0.5) * cubie_width,
345                                   -1 + (j + 0.5) * cubie_width, 0]]
346                                 for i in range(self.N)
347                                 for j in range(self.N)])

350        face_centroids = []

365                                      + translations, rot_mat.T)
