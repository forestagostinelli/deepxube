 53    def __init__(self, dim: int = 7):
 54        super().__init__()
 55        self.dim: int = dim
 56        self.num_tiles: int = self.dim ** 2
 57
 58        self.move_matrix: NDArray = np.zeros((self.num_tiles, 5), dtype=np.int64)
 59        for move in range(self.num_tiles):
 60            x_pos = int(np.floor(move / self.dim))
 61            y_pos = move % self.dim
 62
 63            right = move + self.dim if x_pos < (self.dim-1) else move
 64            left = move - self.dim if x_pos > 0 else move
 65            up = move + 1 if y_pos < (self.dim - 1) else move
 66            down = move - 1 if y_pos > 0 else move
 67
 68            self.move_matrix[move] = [move, right, left, up, down]
 69
 70        self.actions_fixed: List[LOAction] = [LOAction(x) for x in range(self.num_tiles)]
 71        self.goal_np: NDArray = np.zeros(self.num_tiles, dtype=np.uint8)

 74        states_np = np.stack([state.tiles for state in states], axis=0)
 75        goals_np = np.stack([goal.tiles for goal in goals], axis=0)
