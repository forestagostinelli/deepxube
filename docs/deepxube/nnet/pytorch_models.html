 28class SPLASH(nn.Module):
 29    def __init__(self, num_hinges: int = 5, init: str = "RELU"):
 30        super().__init__()
 31        assert num_hinges > 0, "Number of hinges should be greater than zero, but is %s" % num_hinges
 32        assert ((num_hinges + 1) % 2) == 0, "Number of hinges should be odd, but is %s" % num_hinges
 33        init = init.upper()
 34
 35        self.num_hinges: int = num_hinges
 36        self.num_each_side: int = int((self.num_hinges + 1) / 2)
 37
 38        self.hinges: List[float] = list(np.linspace(0, 2.5, self.num_each_side))
 39
 40        self.output_bias: Parameter = Parameter(torch.zeros(1), requires_grad=True)
 41
 42        self.coeffs_right: Parameter
 43        self.coeffs_left: Parameter
 44        if init == "RELU":
 45            self.coeffs_right = Parameter(torch.cat((torch.ones(1), torch.zeros(self.num_each_side - 1))),
 46                                          requires_grad=True)
 47            self.coeffs_left = Parameter(torch.zeros(self.num_each_side), requires_grad=True)
 48        elif init == "LINEAR":
 49            self.coeffs_right = Parameter(torch.cat((torch.ones(1), torch.zeros(self.num_each_side - 1))),
 50                                          requires_grad=True)
 51            self.coeffs_left = Parameter(torch.cat((-torch.ones(1), torch.zeros(self.num_each_side - 1))),
 52                                         requires_grad=True)
 53        else:
 54            raise ValueError("Unknown init %s" % init)
 55
 56    def forward(self, x: Tensor) -> Tensor:
 57        output: Tensor = torch.zeros_like(x)
 58
 59        # output for x > 0
 60        for idx in range(self.num_each_side):
 61            output = output + self.coeffs_right[idx] * torch.clamp(x - self.hinges[idx], min=0)
 62
 63        # output for x < 0
 64        for idx in range(self.num_each_side):
 65            output = output + self.coeffs_left[idx] * torch.clamp(-x - self.hinges[idx], min=0)
 66
 67        output = output + self.output_bias
 68
 69        return output

 72class SPLASH1(nn.Module):

124class ResnetModel(nn.Module):
125    def __init__(self, block_init: Callable[[], nn.Module], num_resnet_blocks: int, act_fn: str):
126        super().__init__()
127        self.blocks: nn.ModuleList = nn.ModuleList()
128        self.act_fns: nn.ModuleList = nn.ModuleList()
129
130        # resnet blocks
131        for block_num in range(num_resnet_blocks):
132            block_net: nn.Module = block_init()
133            module_list: nn.ModuleList = nn.ModuleList([block_net])
134
135            self.blocks.append(module_list)
136            self.act_fns.append(get_act_fn(act_fn))
137
138    def forward(self, x: Tensor) -> Tensor:
139        # resnet blocks
140        module_list: nn.Module
141        for module_list, act_fn in zip(self.blocks, self.act_fns):
142            assert isinstance(module_list, nn.ModuleList)
143            res_inp = x
144            for module in module_list:
145                x = module(x)
146
147            x = act_fn(x + res_inp)
148
149        return x

152class FullyConnectedModel(nn.Module):
153    def __init__(self, input_dim: int, dims: List[int], acts: List[str], batch_norms: Optional[List[bool]] = None,
154                 weight_norms: Optional[List[bool]] = None, group_norms: Optional[List[int]] = None):
155        super().__init__()
156        if batch_norms is None:
157            batch_norms = [False] * len(dims)
158        if weight_norms is None:
159            weight_norms = [False] * len(dims)
160        if group_norms is None:
161            group_norms = [-1] * len(dims)
162        self.layers: nn.ModuleList = nn.ModuleList()
163
164        # layers
165        for dim, act, batch_norm, weight_norm, group_norm in zip(dims, acts, batch_norms, weight_norms, group_norms,
166                                                                 strict=True):
167            module_list = nn.ModuleList()
168
169            # linear
170            if weight_norm:
171                module_list.append(nn.utils.parametrizations.weight_norm(nn.Linear(input_dim, dim)))
172            else:
173                module_list.append(nn.Linear(input_dim, dim))
174
175            # batch norm
176            if batch_norm:
177                module_list.append(nn.BatchNorm1d(dim))
178
179            # group norm
180            if group_norm > 0:
181                module_list.append(nn.GroupNorm(group_norm, dim))
182
183            # activation
184            module_list.append(get_act_fn(act))
185            self.layers.append(module_list)
186
187            input_dim = dim
188
189    def forward(self, x: Tensor) -> Tensor:
190        x = x.float()
191
192        module_list: nn.Module
193        for module_list in self.layers:
194            assert isinstance(module_list, nn.ModuleList)
195            for module in module_list:
196                x = module(x)
197
198        return x

202    def _forward_unimplemented(self, *input_val: Any) -> None:
203        pass

205    def __init__(self, chan_in: int, channel_sizes: List[int], kernel_sizes: List[int], paddings: List[int],
206                 layer_acts: List[str], batch_norms: Optional[List[bool]] = None, strides: Optional[List[int]] = None,
207                 transpose: bool = False, weight_norms: Optional[List[bool]] = None,
208                 dropouts: Optional[List[float]] = None):
209        super().__init__()
210        self.layers: nn.ModuleList = nn.ModuleList()
211        if strides is None:
212            strides = [1] * len(channel_sizes)
213
214        if batch_norms is None:
215            batch_norms = [False] * len(channel_sizes)
216
217        if weight_norms is None:
218            weight_norms = [False] * len(channel_sizes)
219
220        if dropouts is None:
221            dropouts = [0.0] * len(channel_sizes)
222
223        # layers
224        for chan_out, kernel_size, padding, batch_norm, act, stride, weight_norm, dropout in \
225                zip(channel_sizes, kernel_sizes, paddings, batch_norms, layer_acts, strides, weight_norms,
226                    dropouts):
227
228            module_list = nn.ModuleList()
229
230            # linear
231            conv_layer: Union[nn.Conv2d, nn.ConvTranspose2d]
232            if transpose:
233                conv_layer = nn.ConvTranspose2d(chan_in, chan_out, kernel_size, padding=padding, stride=stride)
234            else:
235                conv_layer = nn.Conv2d(chan_in, chan_out, kernel_size, padding=padding, stride=stride)
236
237            if weight_norm:
238                conv_layer = parametrizations.weight_norm(conv_layer)
239
240            module_list.append(conv_layer)
241
242            # batch norm
243            if batch_norm:
244                module_list.append(nn.BatchNorm2d(chan_out))
245
246            # activation
247            module_list.append(get_act_fn(act))
248
249            # dropout
250            if dropout > 0.0:
251                module_list.append(nn.Dropout(dropout))
252
253            self.layers.append(module_list)
254
255            chan_in = chan_out
256
257    def forward(self, x: Tensor) -> Tensor:
258        x = x.float()
259
260        module_list: nn.Module
261        for module_list in self.layers:
262            assert isinstance(module_list, nn.ModuleList)
263            for module in module_list:
264                x = module(x)

266        return x
