201    def __hash__(self) -> int:
202        if self.hash is not None:
203            return self.hash
204
205        # make lit nodes
206        lit_nodes: List[LitNode] = []
207        for lit in [self.head]:
208            lit_nodes.append(LitNode(lit.predicate, False, lit.arguments))
209
210        for lit in self.body:
211            lit_nodes.append(LitNode(lit.predicate, True, lit.arguments))
212
213        # make connections between variable nodes
214        for lit1_idx, lit_node1 in enumerate(lit_nodes):
215            for lit2_idx, lit_node2 in enumerate(lit_nodes):
216                if lit1_idx == lit2_idx:
217                    continue
218                for var_idx1, var_name1 in enumerate(lit_node1.var_names):
219                    for var_idx2, var_name2 in enumerate(lit_node2.var_names):
220                        if var_name1 == var_name2:
221                            lit_node1.var_nodes[var_idx1].add_neighbor(lit_node2.var_nodes[var_idx2])
222
223        # get all variable nodes
224        var_nodes: List[VarNode] = []
225        for lit_node in lit_nodes:
226            var_nodes.extend(lit_node.var_nodes)
227
228        # init representation
229        for lit_node in lit_nodes:
230            lit_node.prop_up()
231
232        # propagate
233        self.hash = 0
234        for _ in range(10):
235            for lit_node in lit_nodes:
236                lit_node.prop_down()
237
238            prop_across(var_nodes)
239
240            for lit_node in lit_nodes:
241                lit_node.prop_up()
242
243        self.hash = sum([lit_node.rep for lit_node in lit_nodes])
244
245        return self.hash

276def theta_sub_lits(lits1: List[Literal], lits2_dict: Dict[str, List[Literal]], negate_l: List[bool],
277                   subs_prev: Dict[str, str], subs_forbid: Dict[str, List[str]]) -> Optional[Dict[str, str]]:
278    # TODO handle negative literals
279    if len(lits1) == 0:
280        return subs_prev
281
282    # TODO, negation as failure not really theta subsumption
283    lit1: Literal = lits1[0]
284    negate: bool = negate_l[0]
285
286    lits2: List[Literal] = lits2_dict.get(lit1.predicate, [])
287    if len(lits2) == 0:
288        if negate:
289            return theta_sub_lits(lits1[1:], lits2_dict, negate_l[1:], subs_prev, subs_forbid)
290        else:
291            return None
292
293    idxs_not_subbed: List[int] = []
294    idxs_vars_req: List[Tuple[int, str]] = []
295    for i, x in enumerate(lit1.arguments):
296        var_sub: Optional[str] = subs_prev.get(x)
297        if var_sub is not None:
298            idxs_vars_req.append((i, var_sub))
299        else:
300            idxs_not_subbed.append(i)
301
302    lits2 = [x for x in lits2 if not prune_lit(lit1, x, idxs_vars_req)]
303    if len(lits2) == 0:
304        if negate:
305            return theta_sub_lits(lits1[1:], lits2_dict, negate_l[1:], subs_prev, subs_forbid)
306        else:
307            return None
308
309    subs_rec: Optional[Dict[str, str]]
310    if len(idxs_not_subbed) == 0:
311        # succeeded somewhere and no substitutions needed
312        if negate:
313            return None
314        else:
315            subs_rec = theta_sub_lits(lits1[1:], lits2_dict, negate_l[1:], subs_prev, subs_forbid)
316            if subs_rec is not None:
317                return subs_rec
318    else:
319        for lit2 in lits2:
320            subs: Optional[Dict[str, str]] = theta_sub_args(lit1.arguments, lit2.arguments, idxs_not_subbed,
321                                                            subs_prev, subs_forbid)
322            if subs is not None:
323                if negate:
324                    return None
325                else:
326                    subs_rec = theta_sub_lits(lits1[1:], lits2_dict, negate_l[1:], subs, subs_forbid)
327                    if subs_rec is not None:
328                        return subs_rec
329
330    if negate:
331        return theta_sub_lits(lits1[1:], lits2_dict, negate_l[1:], subs_prev, subs_forbid)
332    else:
333        return None

336def theta_sub_args(args1: Tuple[str, ...], args2: Tuple[str, ...], idxs_not_subbed: List[int],
337                   subs_prev: Dict[str, str], subs_forbid: Dict[str, List[str]]) -> Optional[Dict[str, str]]:
338    # assuming previous subs checked already if not negate
339    subs: Dict[str, str] = subs_prev.copy()
340
341    for idx in idxs_not_subbed:
342        arg1: str = args1[idx]
343        arg2: str = args2[idx]
344
345        arg1_sub: Optional[str] = subs.get(arg1)
346        if (arg1_sub is not None) and (arg1_sub != arg2):
347            return None
348        else:
349            # check if neq constraint not violated
350            args_other: Optional[List[str]] = subs_forbid.get(arg1, [])
351            if args_other is not None:
352                for arg_other in args_other:
353                    arg_other_sub: Optional[str] = subs.get(arg_other)
354                    if (arg_other_sub is not None) and (arg_other_sub == arg2):
355                        return None
356
357            is_var: bool = arg1[0].isupper() and arg1[0].isalpha()
358            if is_var:
359                # is variable, make substitution
360                subs[arg1] = arg2
361            else:
362                # not variable, cannot substitute
363                if arg1 != arg2:
364                    return None
365
366    return subs

382def theta_sub_replace(clause1: Clause, clause2: Clause, ignore_head: bool = False) -> Clause:
383    subs: Optional[Dict[str, str]] = clause1.theta_sub(clause2, ignore_head=ignore_head)
384    if subs is None:
385        return Clause(clause2.head, clause2.body)
386
387    clause1 = make_subs(clause1, subs)
388    body_new: List[Literal] = [clause1.head]
389
390    # TODO changes in to_code() may break this, what about direction?
391    clause1_body_code_set: Set[str] = set(lit.to_code() for lit in clause1.body)
392    body_new = body_new + [lit for lit in clause2.body if lit.to_code() not in clause1_body_code_set]
393
394    clause2_new: Clause = Clause(clause2.head, tuple(body_new))
395    return clause2_new
