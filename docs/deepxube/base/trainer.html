 62class DataBuffer:
 63    def __init__(self, max_size: int, shapes: List[Tuple[int, ...]], dtypes: List[np.dtype]):
 64        self.arrays: List[NDArray] = []
 65        self.max_size: int = max_size
 66        self.curr_size: int = 0
 67        self.add_idx: int = 0
 68
 69        # first add
 70        start_time = time.time()
 71        print(f"Initializing data buffer with max size {format(self.max_size, ',')}")
 72        print("Input array sizes:")
 73        for array_idx, (shape, dtype) in enumerate(zip(shapes, dtypes)):
 74            print(f"index: {array_idx}, dtype: {dtype}, shape:", shape)
 75            array: NDArray = np.empty((self.max_size,) + shape, dtype=dtype)
 76            self.arrays.append(array)
 77
 78        print(f"Data buffer initialized. Time: {time.time() - start_time}")
 79
 80    def add(self, arrays_add: List[NDArray]) -> None:
 81        self.curr_size = min(self.curr_size + arrays_add[0].shape[0], self.max_size)
 82        assert len(self.arrays) > 0, "Data buffer should have at least one array."
 83        self._add_circular(arrays_add)
 84
 85    def sample(self, sel_idxs: NDArray) -> List[NDArray]:
 86        arrays_samp: List[NDArray] = sel_l(self.arrays, sel_idxs)
 87
 88        return arrays_samp
 89
 90    def size(self) -> int:
 91        return self.curr_size
 92
 93    def clear(self) -> None:
 94        self.curr_size = 0
 95        self.add_idx = 0
 96
 97    def _add_circular(self, arrays_add: List[NDArray]) -> None:
 98        start_idx: int = 0
 99        num_add: int = arrays_add[0].shape[0]
100        assert len(self.arrays) == len(arrays_add), "should have same number of arrays"
101        while start_idx < num_add:
102            num_add_i: int = min(num_add - start_idx, self.max_size - self.add_idx)
103            end_idx: int = start_idx + num_add_i
104            add_idx_end: int = self.add_idx + num_add_i
105
106            for input_idx in range(len(self.arrays)):
107                self.arrays[input_idx][self.add_idx:add_idx_end] = arrays_add[input_idx][start_idx:end_idx]
108
109            start_idx = end_idx
110            self.add_idx = add_idx_end
111            if self.add_idx == self.max_size:
112                self.add_idx = 0

115class Status:
116    def __init__(self, step_max: int, balance_steps: bool):
117        self.itr: int = 0
118        self.update_num: int = 0
119        self.targ_update_num: int = 0
120        self.step_max: int = step_max
121        self.step_probs: NDArray
122        self.step_max_curr: int = 1
123        if balance_steps:
124            self.step_probs = np.zeros(self.step_max + 1)
125            self.step_probs[0:2] = 0.5
126            self.step_probs[2:] = 0
127        else:
128            self.step_probs = np.ones(self.step_max + 1)/(self.step_max + 1)
129        self.per_solved_best: float = 0.0
130
131    def update_step_probs(self, step_to_search_perf: Dict[int, PathFindPerf]) -> None:
132        ave_solve: float = float(np.mean([step_to_search_perf[step].per_solved()
133                                          for step in step_to_search_perf.keys()]))
134        if ave_solve >= 50.0:
135            self.step_max_curr = min(self.step_max_curr * 2, self.step_max)
136
137        self.step_probs = np.zeros(self.step_max + 1)
138        self.step_probs[np.arange(0, self.step_max_curr + 1)] = 1 / (self.step_max_curr + 1)

146    def update_pathfindstats(self, step_to_pathfindperf: Dict[int, PathFindPerf], itr: int) -> None:
147        self.itr_to_steps_to_pathfindstats[itr] = dict()
148        for step, pathfindperf in step_to_pathfindperf.items():
149            self.itr_to_steps_to_pathfindstats[itr][step] = dict()

152            self.itr_to_steps_to_pathfindstats[itr][step]["search_itrs"] = pathfindperf.stats()[2]
153            self.itr_to_steps_to_pathfindstats[itr][step]["ctgs_backup"] = float(np.mean(pathfindperf.ctgs_bkup))
154            self.itr_to_steps_to_pathfindstats[itr][step]["num_instances"] = len(pathfindperf.ctgs_bkup)

174    def __init__(self, nnet: NNet, updater: Up, to_main_q: Queue, from_main_qs: List[Queue], nnet_file: str, nnet_targ_file: str, status_file: str,
175                 train_summary_file: str, device: torch.device, on_gpu: bool, writer: SummaryWriter, train_args: TrainArgs) -> None:
176        self.updater: Up = updater
177        self.to_main_q: Queue = to_main_q
178        self.from_main_qs: List[Queue] = from_main_qs
179        self.nnet: NNet = nnet
180        self.nnet_file = nnet_file
181        self.nnet_targ_file: str = nnet_targ_file
182        self.writer: SummaryWriter = writer
183        self.train_args: TrainArgs = train_args
184        self.device: torch.device = device
185        self.on_gpu: bool = on_gpu
186
187        # load status
188        self.status_file: str = status_file
189        self.status: Status
190        if os.path.isfile(self.status_file):
191            self.status = pickle.load(open(self.status_file, "rb"))
192            print(f"Loaded with itr: {self.status.itr}, update_num: {self.status.update_num}, targ_update_num: {self.status.targ_update_num}")
193        else:
194            self.status = Status(self.updater.up_args.step_max, train_args.balance_steps)
195            # noinspection PyTypeChecker
196            pickle.dump(self.status, open(self.status_file, "wb"), protocol=-1)
197
198        self.train_summary_file: str = train_summary_file
199        self.train_summary: TrainSummary
200        if os.path.isfile(self.train_summary_file):
201            self.train_summary = pickle.load(open(self.train_summary_file, "rb"))
202        else:
203            self.train_summary = TrainSummary()
204            # noinspection PyTypeChecker
205            pickle.dump(self.status, open(self.status_file, "wb"), protocol=-1)
206
207        # load nnet
208        if os.path.isfile(self.nnet_file):
209            self.nnet = cast(NNet, nnet_utils.load_nnet(self.nnet_file, self.nnet))
210        else:
211            torch.save(self.nnet.state_dict(), self.nnet_file)
212            if (self.train_args.checkpoint > 0) and (self.status.update_num == 0):
213                self._save_checkpoint()
214
215        if not os.path.isfile(self.nnet_targ_file):
216            torch.save(self.nnet.state_dict(), self.nnet_targ_file)
217        self.optimizer: Optimizer = self.nnet.get_optimizer()
218
219        self.nnet.to(self.device)
220        if self.data_parallel():
221            self.nnet = cast(NNet, nn.DataParallel(self.nnet))
222
223        # init data buffer
224        shapes_dtypes: List[Tuple[Tuple[int, ...], np.dtype]] = self._get_shapes_dtypes()
225        db_shapes: List[Tuple[int, ...]] = [x[0] for x in shapes_dtypes]
226        db_dtypes: List[np.dtype] = [x[1] for x in shapes_dtypes]
227        self.db: DataBuffer = DataBuffer(self.train_args.batch_size * self.updater.up_args.get_up_gen_itrs(), db_shapes, db_dtypes)
228
229        # optimizer and criterion
230        self.train_start_time = time.time()

232    def update_step(self) -> None:
233        self.db.clear()
234        itr_init: int = self.status.itr
235
236        # print info
237        start_info_l: List[str] = [f"itr: {self.status.itr}", f"update_num: {self.status.update_num}", f"targ_update: {self.status.targ_update_num}"]
238
239        num_gen: int = self.train_args.batch_size * self.updater.up_args.get_up_gen_itrs()
240        start_info_l.append(f"num_gen: {format(num_gen, ',')}")
241        if self.train_args.balance_steps:
242            start_info_l.append(f"step max (curr): {self.status.step_max_curr}")
243        print(f"\nGetting Data - {', '.join(start_info_l)}")
244        times: Times = Times()
245
246        # start updater
247        start_time = time.time()
248        self.updater.start_update(self.status.step_probs.tolist(), num_gen, self.train_args.batch_size, self.device, self.on_gpu)
249        times.record_time("up_start", time.time() - start_time)
250
251        # do training
252        self.train_start_time = time.time()
253        loss: float
254        if not self.updater.up_args.sync_main:
255            self._get_update_data(num_gen, times)
256            self._end_update(itr_init, times)
257            loss = self._train(times)
258        else:
259            loss = self._train_sync_main(num_gen, times)
260            self._end_update(itr_init, times)
261
262        # save nnet
263        start_time = time.time()
264        torch.save(self.nnet.state_dict(), self.nnet_file)
265        if (self.train_args.checkpoint > 0) and (self.status.update_num % self.train_args.checkpoint == 0):
266            self._save_checkpoint()
267        times.record_time("save_net", time.time() - start_time)
268
269        # update nnet
270        update_targ: bool = False
271        if loss < self.train_args.loss_thresh:
272            update_targ = True
273
274        if update_targ:
275            shutil.copy(self.nnet_file, self.nnet_targ_file)
276            self.status.targ_update_num = self.status.targ_update_num + 1
277        self.status.update_num += 1
278
279        start_time = time.time()
280        # noinspection PyTypeChecker
281        pickle.dump(self.status, open(self.status_file, "wb"), protocol=-1)
282        # noinspection PyTypeChecker
283        pickle.dump(self.train_summary, open(self.train_summary_file, "wb"), protocol=-1)
284        times.record_time("save_status", time.time() - start_time)
285        print(f"Train - itrs: {self.updater.up_args.up_itrs}, loss: {loss:.2E}, targ_updated: {update_targ}")
286        print(f"Times - {times.get_time_str()}")

288    def _get_update_data(self, num_gen: int, times: Times) -> None:
289        start_time = time.time()
290        while self.db.size() < num_gen:
291            data_l: List[List[NDArray]] = self.updater.get_update_data()
292            for data in data_l:
293                self.db.add(data)
294        times.record_time("up_data", time.time() - start_time)
295
296    def _train(self, times: Times) -> float:
297        loss: float = np.inf
298        first_itr_in_update: bool = True
299        sel_idx_start: int = 0
300        sel_idxs_rand_order: NDArray = np.random.choice(self.db.size(), size=self.db.size(), replace=False)
301        for _ in range(self.updater.up_args.up_itrs):
302            # sample data
303            start_time = time.time()
304            sel_idxs: NDArray = np.arange(sel_idx_start, sel_idx_start + self.train_args.batch_size) % self.db.size()
305
306            batch: List[NDArray] = self.db.sample(sel_idxs_rand_order[sel_idxs])
307            times.record_time("data_samp", time.time() - start_time)
308
309            # train
310            loss = self._train_itr(batch, first_itr_in_update, times)
311            first_itr_in_update = False
312            self.status.itr += 1
313
314            # update sel_idx
315            if sel_idxs.max() == (self.db.size() - 1):
316                sel_idx_start = 0
317                sel_idxs_rand_order = np.random.choice(self.db.size(), size=self.db.size(), replace=False)
318            else:
319                sel_idx_start = int(sel_idxs[-1]) + 1
320
321        return loss
322
323    def _train_sync_main(self, num_gen: int, times: Times) -> float:
324        loss: float = np.inf
325        update_train_itr: int = 0
326        first_itr_in_update: bool = True
327        while update_train_itr < self.updater.up_args.up_itrs:
328            # data from updater should not be more that train_args.batch_size
329            start_time = time.time()
330            sel_idxs: NDArray
331            if self.db.size() == num_gen:
332                sel_idxs = np.random.randint(self.db.size(), size=self.train_args.batch_size)
333            else:
334                # compute heuristic values for ongoing search and get data
335                self.nnet.eval()
336                while self.db.size() < ((update_train_itr + 1) * self.train_args.batch_size):
337                    # compute heuristic values
338                    q_res: Optional[Tuple[int, List[SharedNDArray]]] = get_nowait_noerr(self.to_main_q)
339                    if q_res is not None:
340                        proc_id, inputs_np_shm = q_res
341                        nnet_in_out_shared_q(self.nnet, inputs_np_shm, self.updater.up_args.nnet_batch_size,
342                                             self.device, self.from_main_qs[proc_id])
343
344                    # get update data
345                    data_l_i: List[List[NDArray]] = self.updater.get_update_data(nowait=True)
346                    for data in data_l_i:
347                        self.db.add(data)
348
349                # select incides
350                sel_idxs = np.arange(update_train_itr * self.train_args.batch_size, (update_train_itr + 1) * self.train_args.batch_size)
351
352            batch: List[NDArray] = self.db.sample(sel_idxs)
353
354            times.record_time("up_data", time.time() - start_time)
355
356            # train
357            loss = self._train_itr(batch, first_itr_in_update, times)
358            update_train_itr += 1
359            first_itr_in_update = False
360            self.status.itr += 1
361
362        return loss

368    def _end_update(self, itr_init: int, times: Times) -> None:
369        start_time = time.time()
370        step_to_search_perf: Dict[int, PathFindPerf] = self.updater.end_update()
371        self.train_summary.update_pathfindstats(step_to_search_perf, itr_init)
372        if self.train_args.balance_steps:
373            self.status.update_step_probs(step_to_search_perf)
374
375        per_solved_ave, path_costs_ave, search_itrs_ave = get_eq_weighted_perf(step_to_search_perf)
376
377        self.writer.add_scalar("train/pathfind/solved", per_solved_ave, self.status.itr)
378        self.writer.add_scalar("train/pathfind/path_cost", path_costs_ave, self.status.itr)
379        self.writer.add_scalar("train/pathfind/search_itrs", search_itrs_ave, self.status.itr)
380
381        post_up_info_l: List[str] = [f"%solved: {per_solved_ave:.2f}", f"path_costs: {path_costs_ave:.3f}",
382                                     f"search_itrs: {search_itrs_ave:.3f}"] + self._add_post_up_info()
383
384        print(f"Data - {', '.join(post_up_info_l)}")
385
386        times.record_time("up_end", time.time() - start_time)

256            self._get_update_data(num_gen, times)
257            self._end_update(itr_init, times)
258            loss = self._train(times)
259        else:
260            loss = self._train_sync_main(num_gen, times)
261            self._end_update(itr_init, times)
262
263        # save nnet
264        start_time = time.time()
265        torch.save(self.nnet.state_dict(), self.nnet_file)
266        if (self.train_args.checkpoint > 0) and (self.status.update_num % self.train_args.checkpoint == 0):
267            self._save_checkpoint()
268        times.record_time("save_net", time.time() - start_time)
269
270        # update nnet
271        update_targ: bool = False
272        if loss < self.train_args.loss_thresh:
273            update_targ = True
274
275        if update_targ:
276            shutil.copy(self.nnet_file, self.nnet_targ_file)
277            self.status.targ_update_num = self.status.targ_update_num + 1
278        self.status.update_num += 1
279
280        start_time = time.time()
281        # noinspection PyTypeChecker
282        pickle.dump(self.status, open(self.status_file, "wb"), protocol=-1)
283        # noinspection PyTypeChecker
284        pickle.dump(self.train_summary, open(self.train_summary_file, "wb"), protocol=-1)
285        times.record_time("save_status", time.time() - start_time)
286        print(f"Train - itrs: {self.updater.up_args.up_itrs}, loss: {loss:.2E}, targ_updated: {update_targ}")
287        print(f"Times - {times.get_time_str()}")
