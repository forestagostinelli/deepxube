 41    def bellman_backup(self) -> float:
 42        assert self.is_solved is not None
 43
 44        if self.is_solved:
 45            self.backup_val = 0.0
 46        else:
 47            if len(self.edge_dict) > 0:
 48                self.backup_val = min(tc + node_c.heuristic for tc, node_c in self.edge_dict.values())
 49        return self.backup_val

 51    def upper_bound_parent_path(self, ctg_ub: float) -> None:
 52        self.backup_val = min(self.backup_val, ctg_ub)
 53        if self.parent is not None:
 54            assert self.parent_t_cost is not None
 55            self.parent.upper_bound_parent_path(ctg_ub + self.parent_t_cost)

 57    def tree_backup(self) -> float:
 58        if (self.is_solved is not None) and self.is_solved:
 59            self.backup_val = 0.0
 60        else:
 61            if len(self.edge_dict) == 0:
 62                self.backup_val = max(self.heuristic, 0.0)
 63            else:
 64                self.backup_val = min(tc + node_c.tree_backup() for tc, node_c in self.edge_dict.values())
 65
 66        return self.backup_val

 68    def backup_act(self, action: Action) -> float:
 69        assert self.is_solved is not None
 70        if self.is_solved:
 71            return 0.0
 72        else:
 73            tc, node_next = self.edge_dict[action]
 74            # assert node_next.q_values is not None
 75            if node_next.backup_val < np.inf:
 76                return tc + node_next.backup_val
 77            else:
 78                return tc + node_next.heuristic

 80    def get_all_descendants(self) -> List['Node']:
 81        """ Get all descendants of node (excluding self)
 82
 83        :return: List of nodes that are descendants
 84        """
 85        fifo: List[Node] = [x[1] for x in self.edge_dict.values()]
 86        descendants: List[Node] = []
 87        while len(fifo) > 0:
 88            descendant: Node = fifo.pop(0)
 89            for _, descendant_c in descendant.edge_dict.values():
 90                fifo.append(descendant_c)
 91            descendants.append(descendant)
 92        return descendants

 95def get_path(node: Node) -> Tuple[List[State], List[Action], List[float], float]:
 96    """ Gets path from the start state to the goal state associated with the input node
 97
 98    :param node: goal node
 99    :return: List of states along path, List of actions in path, path cost
100    """
101    path: List[State] = []
102    actions: List[Action] = []
103    tcs: List[float] = []
104
105    parent_node: Node = node
106    while parent_node.parent is not None:
107        path.append(parent_node.state)
108
109        assert parent_node.parent_action is not None, "parent_action should not be None"
110        assert parent_node.parent_t_cost is not None, "parent_t_cost should not be None"
111        actions.append(parent_node.parent_action)
112        tcs.append(parent_node.parent_t_cost)
113        parent_node = parent_node.parent
114
115    path.append(parent_node.state)
116
117    path = path[::-1]
118    actions = actions[::-1]
119    tcs = tcs[::-1]
120
121    assert sum(tcs) == node.path_cost, "sum of transition costs should equal path cost"
122    return path, actions, tcs, node.path_cost

378    def step(self, verbose: bool = False) -> Tuple[List[Node], List[EdgeQ]]:
379        instances: List[INode] = [instance for instance in self.instances if not instance.finished()]
380        if len(instances) == 0:
381            self.itr += 1  # TODO make more elegant
382            return [], []
383
384        # pop from open
385        start_time = time.time()
386        nodes_popped_by_inst: List[List[Node]] = [instance.get_nodes() for instance in instances]
387        nodes_popped_flat: List[Node] = misc_utils.flatten(nodes_popped_by_inst)[0]
388        self.times.record_time("pop", time.time() - start_time)
389
390        # is solved
391        self.set_is_solved(nodes_popped_flat)
392        # for node in nodes_popped_flat:
393        #    print(node.state, node.is_solved)
394
395        # record goal
396        start_time = time.time()
397        for instance, nodes in zip(instances, nodes_popped_by_inst, strict=True):
398            instance.record_goal(nodes)
399        self.times.record_time("goal", time.time() - start_time)
400
401        # expand
402        nodes_exp_by_inst: List[List[Node]] = self._expand(instances, nodes_popped_by_inst)
403
404        # eval nodes
405        self._set_node_vals(misc_utils.flatten(nodes_exp_by_inst)[0])
406
407        # filter expanded nodes
408        start_time = time.time()
409        for inst_idx, instance in enumerate(instances):
410            nodes_exp_by_inst[inst_idx] = instance.filter_expanded_nodes(nodes_exp_by_inst[inst_idx])
411        self.times.record_time("filt", time.time() - start_time)
412
413        # get costs
414        costs_by_inst: List[List[float]] = self._compute_costs(instances, nodes_exp_by_inst)
415
416        # push
417        start_time = time.time()
418        nodes_next_by_inst: List[List[Node]] = []
419        for instance, nodes_exp, costs in zip(instances, nodes_exp_by_inst, costs_by_inst, strict=True):
420            nodes_next_by_inst.append(instance.push_pop_nodes(nodes_exp, costs))
421        self.times.record_time("pushpop", time.time() - start_time)
422
423        # get next edges
424        start_time = time.time()
425        edges_next_flat: List[EdgeQ] = []
426        for instance, nodes_next in zip(instances, nodes_next_by_inst, strict=True):
427            edges_popped_inst: List[EdgeQ] = []
428            for node in nodes_next:
429                assert (node.parent is not None) and (node.parent_action is not None) and (node.parent_t_cost is not None)
430                edges_popped_inst.append(EdgeQ(node.parent, node.parent_action, node.parent_t_cost + node.heuristic))
431            instance.add_edges_popped(edges_popped_inst)
432            edges_next_flat.extend(edges_popped_inst)
433        self.times.record_time("edges_next", time.time() - start_time)
434
435        start_time = time.time()
436        for instance, nodes_next in zip(instances, nodes_next_by_inst, strict=True):
437            instance.set_next_nodes(nodes_next)
438        self.times.record_time("set_next", time.time() - start_time)
439
440        # update iterations
441        self.itr += 1
442        for instance in instances:
443            instance.itr += 1
444
445        # verbose
446        if verbose:
447            self._verbose(instances, nodes_popped_by_inst)
448
449        return nodes_popped_flat, edges_next_flat

451    def _expand(self, instances: List[INode], nodes_by_inst: List[List[Node]]) -> List[List[Node]]:
452        start_time = time.time()
453        # flatten (for speed)
454        nodes: List[Node]
455        split_idxs: List[int]
456        nodes, split_idxs = misc_utils.flatten(nodes_by_inst)
457
458        if len(nodes) == 0:
459            return [[]]
460
461        # Get children of nodes
462        states: List[State] = [x.state for x in nodes]
463        goals: List[Goal] = [x.goal for x in nodes]
464
465        states_c_l: List[List[State]]
466        actions: List[List[Action]]
467        tcs: List[List[float]]
468        states_c_l, actions, tcs = self.expand_states(states, goals)
469
470        goals_c: List[List[Goal]] = [[node.goal] * len(state_c) for node, state_c in zip(nodes, states_c_l, strict=True)]
471        states_c_flat: List[State]
472        states_c_flat, split_idxs_c = misc_utils.flatten(states_c_l)
473        goals_c_flat, _ = misc_utils.flatten(goals_c)
474        self.times.record_time("expand", time.time() - start_time)
475
476        # get children nodes
477        start_time = time.time()
478        nodes_c: List[Node] = []
479        for node_idx, node in enumerate(nodes):
480            path_costs_c_i: NDArray = node.path_cost + np.array(tcs[node_idx])
481            nodes_c_i: List[Node] = []
482            for c_idx in range(len(states_c_l[node_idx])):
483                action: Action = actions[node_idx][c_idx]
484
485                edge_dict_val: Optional[Tuple[float, Node]] = node.edge_dict.get(action)
486                node_c: Node
487                if edge_dict_val is not None:
488                    node_c = edge_dict_val[1]
489                else:
490                    t_cost: float = tcs[node_idx][c_idx]
491                    node_c = Node(states_c_l[node_idx][c_idx], goals_c[node_idx][c_idx], float(path_costs_c_i[c_idx]), 0.0, None, None, action, t_cost, node)
492                    node.add_edge(action, t_cost, node_c)
493
494                nodes_c_i.append(node_c)
495            nodes_c.extend(nodes_c_i)
496        self.times.record_time("nodes", time.time() - start_time)
497
498        # get child nodes by instance
499        start_time = time.time()
500        nodes_c_by_state: List[List[Node]] = misc_utils.unflatten(nodes_c, split_idxs_c)
501        nodes_c_by_inst_state: List[List[List[Node]]] = misc_utils.unflatten(nodes_c_by_state, split_idxs)
502        nodes_c_by_inst: List[List[Node]] = []
503        for nodes_c_by_inst_state_i in nodes_c_by_inst_state:
504            nodes_c_by_inst.append(misc_utils.flatten(nodes_c_by_inst_state_i)[0])
505
506        for instance, nodes_by_inst_i, nodes_c_by_inst_i in zip(instances, nodes_by_inst, nodes_c_by_inst, strict=True):
507            instance.add_nodes_popped(nodes_by_inst_i)
508            instance.num_nodes_generated += len(nodes_c_by_inst_i)
509
510        self.times.record_time("up_inst", time.time() - start_time)
511
512        return nodes_c_by_inst
