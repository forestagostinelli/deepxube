119def get_path(node: Node) -> Tuple[List[State], List[Action], float]:
120    """ Gets path from the start state to the goal state associated with the input node
121
122    :param node: goal node
123    :return: List of states along path, List of actions in path, path cost
124    """
125    path: List[State] = []
126    actions: List[Action] = []
127
128    parent_node: Node = node
129    while parent_node.parent is not None:
130        path.append(parent_node.state)
131
132        assert parent_node.parent_action is not None, "parent_action should not be None"
133        actions.append(parent_node.parent_action)
134        parent_node = parent_node.parent
135
136    path.append(parent_node.state)
137
138    path = path[::-1]
139    actions = actions[::-1]
140
141    return path, actions, node.path_cost

163    def bellman_backup(self) -> float:
164        assert self.is_solved is not None
165
166        if self.is_solved:
167            self.backup_val = 0.0
168        else:
169            assert self.children is not None
170            if len(self.children) == 0:
171                self.backup_val = self.heuristic
172            else:
173                assert self.t_costs is not None
174
175                self.backup_val = np.inf
176                for tc, node_c in zip(self.t_costs, self.children, strict=True):
177                    assert node_c.heuristic is not None
178                    self.backup_val = min(self.backup_val, tc + node_c.heuristic)
179
180        return self.backup_val

182    def upper_bound_parent_path(self, ctg_ub: float) -> None:
183        assert self.backup_val is not None
184        self.backup_val = min(self.backup_val, ctg_ub)
185        if self.parent is not None:
186            assert self.parent_t_cost is not None
187            self.parent.upper_bound_parent_path(ctg_ub + self.parent_t_cost)

189    def tree_backup(self) -> float:
190        if (self.is_solved is not None) and self.is_solved:
191            self.backup_val = 0.0
192        else:
193            assert self.heuristic is not None
194            if (self.children is None) or (len(self.children) == 0):
195                self.backup_val = max(self.heuristic, 0.0)
196            else:
197                assert self.children is not None
198                assert self.t_costs is not None
199                self.backup_val = np.inf
200                for tc, node_c in zip(self.t_costs, self.children, strict=True):
201                    self.backup_val = min(self.backup_val, tc + node_c.tree_backup())
