175def get_path(node: Node) -> Tuple[List[State], List[Action], float]:
176    """ Gets path from the start state to the goal state associated with the input node
177
178    :param node: goal node
179    :return: List of states along path, List of actions in path, path cost
180    """
181    path: List[State] = []
182    actions: List[Action] = []
183
184    parent_node: Node = node
185    while parent_node.parent is not None:
186        path.append(parent_node.state)
187
188        assert parent_node.parent_action is not None, "parent_action should not be None"
189        actions.append(parent_node.parent_action)
190        parent_node = parent_node.parent
191
192    path.append(parent_node.state)
193
194    path = path[::-1]
195    actions = actions[::-1]
196
197    return path, actions, node.path_cost

352    def get_next_nodes(self, instances: List[I], edges_by_inst: List[List[EdgeQ]]) -> List[List[Node]]:
353        if len(instances) == 0:
354            return []
355        start_time = time.time()
356        # flatten
357        edges, split_idxs = misc_utils.flatten(edges_by_inst)
358        nodes: List[Node] = [edge.node for edge in edges]
359
360        states: List[State] = [node.state for node in nodes]
361        goals: List[Goal] = [node.goal for node in nodes]
362        actions: List[Optional[Action]] = [node_act.action for node_act in edges]
363        path_costs: List[float] = [popped_node.path_cost for popped_node in nodes]
364
365        # next states
366        states_next = states.copy()
367        tcs: List[float] = [0.0] * len(states_next)
368        idxs_op: List[int] = [idx for idx, action in enumerate(actions) if action is not None]
369        if len(idxs_op) > 0:
370            states_op: List[State] = [states[idx_op] for idx_op in idxs_op]
371            actions_op: List[Action] = [cast(Action, actions[idx_op]) for idx_op in idxs_op]
372            states_next_op, tcs_op = self.domain.next_state(states_op, actions_op)
373            for idx_op, state, tc in zip(idxs_op, states_next_op, tcs_op):
374                states_next[idx_op] = state
375                tcs[idx_op] = tc
376        path_costs_next: List[float] = (np.array(path_costs) + np.array(tcs)).tolist()
377        self.times.record_time("next_state", time.time() - start_time)
378
379        # heuristic function
380        start_time = time.time()
381        q_vals_next, actions_next_l = self.get_qvals_acts(states_next, goals)
382        heurs_next: List[float] = [min(x) for x in q_vals_next]
383        self.times.record_time("heur", time.time() - start_time)
384
385        # next nodes
386        start_time = time.time()
387        nodes_next: List[Node] = []
388        for idx in range(len(edges)):
389            node_next: Node
390            action_i: Optional[Action] = actions[idx]
391            if action_i is not None:
392                node_next = Node(states_next[idx], goals[idx], path_costs_next[idx], heurs_next[idx], (actions_next_l[idx], q_vals_next[idx]), None, action_i,
393                                 tcs[idx], nodes[idx])
394                nodes[idx].add_edge(action_i, tcs[idx], node_next)
395            else:
396                node_next = nodes[idx]
397                node_next.q_values = (actions_next_l[idx], q_vals_next[idx])
398            nodes_next.append(node_next)
399        self.times.record_time("make_nodes", time.time() - start_time)
400
401        # updater instances
402        start_time = time.time()
403        nodes_next_by_inst: List[List[Node]] = misc_utils.unflatten(nodes_next, split_idxs)
404        for instance, nodes_next_by_inst_i in zip(instances, nodes_next_by_inst, strict=True):
405            instance.num_nodes_generated += len(nodes_next_by_inst_i)
406        self.times.record_time("up_inst", time.time() - start_time)
407
408        return nodes_next_by_inst

411        actions_l: List[List[Action]] = self._get_actions(states, goals)

415    def create_root_nodes(self, states: List[State], goals: List[Goal], compute_init_heur: bool = True) -> List[Node]:
416        start_time = time.time()
417
418        qvals_l: List[List[float]]
419        actions_l: List[List[Action]]
420        heuristics: List[float]
421        if compute_init_heur:
422            qvals_l, actions_l = self.get_qvals_acts(states, goals)
423            heuristics = [min(x) for x in qvals_l]

427            heuristics = [0.0 for _ in states]

430        for state, goal, heuristic, actions, qvals in zip(states, goals, heuristics, actions_l, qvals_l, strict=True):
431            root_node: Node = Node(state, goal, 0.0, heuristic, (actions, qvals), None, None, None, None)
432            root_nodes.append(root_node)
433        self.times.record_time("root", time.time() - start_time)
