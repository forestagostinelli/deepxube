 98class HeurNNet(DeepXubeNNet[In]):
 99    def __init__(self, nnet_input: In, out_dim: int, q_fix: bool, **kwargs: Any):
100        """
101
102        :param nnet_input: Neural network input
103        :param out_dim: Output dimensionality. If q_fix is true, this is the dimensionality of the number of actions, must be 1 otherwise
104        :param q_fix: If true, the last element in the list of numpy arrays of the input corresponds to the index of the action
105        :param kwargs: kwargs
106        """
107        super().__init__(nnet_input)
108        self.out_dim: int = out_dim
109        self.q_fix: bool = q_fix
110
111    def get_loss_and_info(self, fwd_tr_tensors: List[Tensor], get_info: bool) -> Tuple[Tensor, Optional[str]]:
112        assert len(fwd_tr_tensors) == 2
113        ctgs_nnet: Tensor = fwd_tr_tensors[0]
114        ctgs_targ: Tensor = fwd_tr_tensors[1]
115
116        assert ctgs_nnet.size() == ctgs_targ.size()
117
118        loss_mse: Tensor = torch.mean((ctgs_targ - ctgs_nnet) ** 2)
119
120        info: Optional[str] = None
121        if get_info:
122            info = f"targ_ctg: {ctgs_targ.mean().item():.2f}, nnet_ctg: {ctgs_nnet.mean().item():.2f}"
123
124        return loss_mse, info
125
126    def _forward_train(self, inputs: List[Tensor]) -> List[Tensor]:
127        ctgs_targ: Tensor = inputs.pop(-1)
128        ctgs_nnet: Tensor = self._forward_heur(inputs)
129
130        return [ctgs_nnet, ctgs_targ]
131
132    def _forward_eval(self, inputs: List[Tensor]) -> List[Tensor]:
133        return [self._forward_heur(inputs)]
134
135    def _forward_heur(self, inputs: List[Tensor]) -> Tensor:
136        if self.q_fix:
137            action_idxs: Tensor = inputs[-1].long()
138            x: Tensor = self._forward(inputs[:-1])
139            return torch.gather(x, 1, action_idxs)
140        else:
141            return self._forward(inputs)
142
143    @abstractmethod
144    def _forward(self, inputs: List[Tensor]) -> Tensor:
145        pass

165class PolicyVAE(PolicyNNet[PNNetIn]):
166    def __init__(self, nnet_input: PNNetIn, num_samp: int, kl_weight: float, **kwargs: Any):
167        super().__init__(nnet_input, num_samp)
168        self.norm_dist = torch.distributions.Normal(0, 1)
169        self.kl_weight: float = kl_weight
170
171    def get_loss_and_info(self, fwd_tr_tensors: List[Tensor], get_info: bool) -> Tuple[Tensor, Optional[str]]:
172        loss_recon_mean: Tensor = torch.mean(fwd_tr_tensors[0], dim=0)
173        loss_kl_mean: Tensor = torch.mean(fwd_tr_tensors[1], dim=0)
174
175        loss: Tensor = loss_recon_mean + (self.kl_weight * loss_kl_mean)
176
177        loss_str: Optional[str] = None
178        if get_info:
179            loss_str = f"loss_recon: {loss_recon_mean.item():.2E}, loss_kl: {loss_kl_mean.item():.2E}"
180
181        return loss, loss_str
182
183    def _forward_eval(self, states_goals: List[Tensor]) -> List[Tensor]:
184        recons_l: List[List[Tensor]] = []
185        z_l: List[Tensor] = []
186        for _ in range(self.num_samp):
187            z: Tensor = self.norm_dist.sample((states_goals[0].shape[0],) + self.latent_shape()).to(states_goals[0].device)
188            recons: List[Tensor] = self.decode(states_goals, z)
189            recons_l.append(recons)
190            z_l.append(z)
191
192        recons_all: List[Tensor] = []
193        for recon_idx in range(len(recons_l[0])):
194            recons_i: Tensor = torch.stack([recons_i[recon_idx] for recons_i in recons_l], dim=1)
195            recons_all.append(recons_i)
196
197        z_all: Tensor = torch.stack(z_l, dim=1)
198        return recons_all + [self.norm_dist.log_prob(z_all).sum(dim=2)]
199
200    def _forward_train(self, states_goals_actions: List[Tensor]) -> List[Tensor]:
201        split_idx: int = self.nnet_input.states_goals_actions_split_idx()
202        states_goals: List[Tensor] = states_goals_actions[:split_idx]
203        actions: List[Tensor] = states_goals_actions[split_idx:]
204
205        actions_proc, mu, logvar = self.encode(states_goals, actions)
206        sum_dims: Tuple[int, ...] = tuple(range(1, len(mu.shape)))
207        loss_kl: Tensor = -0.5 * torch.sum(1 + logvar - mu ** 2 - logvar.exp(), dim=sum_dims)
208
209        sigma = torch.exp(logvar / 2.0)
210        z = mu + sigma * self.norm_dist.sample(mu.shape).to(mu.device)
211        actions_recon: List[Tensor] = self.decode(states_goals, z)
212
213        loss_recon: Tensor = self._compute_recon_loss(actions_proc, actions_recon)
214
215        return [loss_recon, loss_kl]
216

226    def encode(self, states_goals: List[Tensor], actions: List[Tensor]) -> Tuple[List[Tensor], Tensor, Tensor]:
227        """ Conditon on states and goals and map actions to mu and logvar
228

236        """ Conditon on states and goals and map sampled latent to reconstructed actions
237
238        :param states_goals:
239        :param z: Latent state
240        :return:
241        """
242
243    def _compute_recon_loss(self, action_proc: List[Tensor], actions_recon: List[Tensor]) -> Tensor:
244        loss_recons: List[Tensor] = []
245        for actions_proc_i, actions_recon_i in zip(action_proc, actions_recon):
246            mean_dims: Tuple[int, ...] = tuple(range(1, len(actions_proc_i.shape)))
247            loss_recon_i: Tensor = torch.mean((actions_recon_i - actions_proc_i) ** 2, dim=mean_dims)
248            loss_recons.append(loss_recon_i)
249
250        return torch.stack(loss_recons, dim=0).mean(dim=0)
251
252    def __repr__(self) -> str:
253        return f"{super().__repr__()}\nKL Weight: {self.kl_weight}"
