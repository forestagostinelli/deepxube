 92        info: Optional[str] = None

 99        ctgs_targ: Tensor = inputs.pop(-1)

137class PolicyVAE(PolicyNNet[PNNetIn]):
138    def __init__(self, nnet_input: PNNetIn, num_samp: int, kl_weight: float, **kwargs: Any):
139        super().__init__(nnet_input, num_samp)
140        self.norm_dist = torch.distributions.Normal(0, 1)
141        self.kl_weight: float = kl_weight
142
143    def get_loss_and_info(self, fwd_tr_tensors: List[Tensor], get_info: bool) -> Tuple[Tensor, Optional[str]]:
144        loss_recon_mean: Tensor = torch.mean(fwd_tr_tensors[0], dim=0)
145        loss_kl_mean: Tensor = torch.mean(fwd_tr_tensors[1], dim=0)
146
147        loss: Tensor = loss_recon_mean + (self.kl_weight * loss_kl_mean)
148
149        loss_str: Optional[str] = None
150        if get_info:
151            loss_str = f"loss_recon: {loss_recon_mean.item():.2E}, loss_kl: {loss_kl_mean.item():.2E}"
152
153        return loss, loss_str
154
155    def _forward_eval(self, states_goals: List[Tensor]) -> List[Tensor]:
156        recons_l: List[List[Tensor]] = []
157        z_l: List[Tensor] = []
158        for _ in range(self.num_samp):
159            z: Tensor = self.norm_dist.sample((states_goals[0].shape[0],) + self.latent_shape()).to(states_goals[0].device)
160            recons: List[Tensor] = self.decode(states_goals, z)
161            recons_l.append(recons)
162            z_l.append(z)
163
164        recons_all: List[Tensor] = []
165        for recon_idx in range(len(recons_l[0])):
166            recons_i: Tensor = torch.stack([recons_i[recon_idx] for recons_i in recons_l], dim=1)
167            recons_all.append(recons_i)
168
169        z_all: Tensor = torch.stack(z_l, dim=1)
170        return recons_all + [self.norm_dist.log_prob(z_all).sum(dim=2)]
171
172    def _forward_train(self, states_goals_actions: List[Tensor]) -> List[Tensor]:
173        split_idx: int = self.nnet_input.states_goals_actions_split_idx()
174        states_goals: List[Tensor] = states_goals_actions[:split_idx]
175        actions: List[Tensor] = states_goals_actions[split_idx:]
176
177        actions_proc, mu, logvar = self.encode(states_goals, actions)
178        sum_dims: Tuple[int, ...] = tuple(range(1, len(mu.shape)))
179        loss_kl: Tensor = -0.5 * torch.sum(1 + logvar - mu ** 2 - logvar.exp(), dim=sum_dims)
180
181        sigma = torch.exp(logvar / 2.0)
182        z = mu + sigma * self.norm_dist.sample(mu.shape).to(mu.device)
183        actions_recon: List[Tensor] = self.decode(states_goals, z)
184
185        loss_recon: Tensor = self._compute_recon_loss(actions_proc, actions_recon)
186
187        return [loss_recon, loss_kl]

211    def _compute_recon_loss(self, action_proc: List[Tensor], actions_recon: List[Tensor]) -> Tensor:
212        loss_recons: List[Tensor] = []
213        for actions_proc_i, actions_recon_i in zip(action_proc, actions_recon):
214            mean_dims: Tuple[int, ...] = tuple(range(1, len(actions_proc_i.shape)))
215            loss_recon_i: Tensor = torch.mean((actions_recon_i - actions_proc_i) ** 2, dim=mean_dims)
216            loss_recons.append(loss_recon_i)
