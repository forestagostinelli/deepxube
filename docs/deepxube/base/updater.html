161    def start_procs(self) -> Tuple[Queue, List[Queue]]:
162        # start updater procs
163        # TODO implement safer copy?
164        updaters: List[Update] = [copy.deepcopy(self) for _ in range(self.up_args.procs)]
165
166        # parallel heuristic functions
167        ctx = get_context("spawn")
168        to_main_q: Queue = ctx.Queue()
169        self.from_main_qs = []
170        self.set_nnet_par_info_l_dict()
171        for proc_idx, updater in enumerate(updaters):
172            from_main_q: Queue = ctx.Queue(1)
173            self.from_main_qs.append(from_main_q)
174            updater.set_main_qs(to_main_q, from_main_q, proc_idx)
175            for nnet_name in self.nnet_par_info_l_dict.keys():
176                updater.set_nnet_par_info(nnet_name, self.nnet_par_info_l_dict[nnet_name][proc_idx])
177
178        self.to_q = ctx.Queue()
179        self.from_q = ctx.Queue()
180        self.procs = []
181
182        for updater in updaters:
183            proc: BaseProcess = ctx.Process(target=updater.update_runner, args=(self.to_q, self.from_q))
184            proc.daemon = True
185            proc.start()
186            self.procs.append(proc)

190    def start_update(self, step_probs: List[int], num_gen: int, targ_update_num: Optional[int], train_batch_size: int,
191                     device: torch.device, on_gpu: bool) -> None:
192        # start parallel nnet runners
193        self.set_targ_update_num(targ_update_num)
194        self.start_nnet_runners(device, on_gpu)
195
196        # put update data
197        for proc_idx, from_main_q in enumerate(self.from_main_qs):
198            from_main_q.put((step_probs, targ_update_num))
199
200        # put work information on to_q
201        assert self.to_q is not None
202
203        num_searches: int = num_gen // self.up_args.search_itrs
204        if self.up_args.v:
205            print(f"Generating {format(num_gen, ',')} training instances with {format(num_searches, ',')} searches")
206
207        assert num_gen % self.up_args.search_itrs == 0, (f"Number of instances to generate per for this updater {num_gen} is not divisible by the max number "
208                                                         f"of pathfinding iterations to take during the updater ({self.up_args.search_itrs})")
209        up_batch_size: int = train_batch_size if (self.up_args.up_batch_size is None) else self.up_args.up_batch_size
210        num_to_send_per: List[int] = split_evenly_w_max(num_searches, self.up_args.procs,
211                                                        min(up_batch_size, train_batch_size))
212        for num_to_send_per_i in num_to_send_per:
213            if num_to_send_per_i > 0:
214                self.to_q.put(num_to_send_per_i)

244    def end_update(self) -> Dict[int, PathFindPerf]:
245        assert (self.to_q is not None) and (self.from_q is not None)
246        # sending stop signal
247        for _ in self.procs:

250        # get summary from processes
251        step_to_pathperf: Dict[int, PathFindPerf] = dict()
252        times_up: Times = Times()
253        for _ in self.procs:
254            times_up_i, step_to_pathperf_i = self.from_q.get()
255            times_up.add_times(times_up_i)
256            for step_num_perf, pathperf in step_to_pathperf_i.items():

258                    step_to_pathperf[step_num_perf] = PathFindPerf()
259                step_to_pathperf[step_num_perf] = step_to_pathperf[step_num_perf].comb_perf(pathperf)
260
261        # print
262        if self.up_args.v:
263            print(f"Generated {format(self.num_generated, ',')} training instances")
264            print(f"Times - {times_up.get_time_str()}")
265            print_pathfindperf(step_to_pathperf)
266
267        # clean up clean up everybody do your share
268        for nnet_name, nnet_par_infos in self.nnet_par_info_l_dict.items():

324    def update_runner(self, to_q: Queue, from_q: Queue) -> None:
325        while True:
326            assert self.from_main_q is not None
327            data_q: Optional[Tuple[List[int], Optional[int]]] = self.from_main_q.get()
328            if data_q is None:
329                break
330            times: Times = Times()
331
332            step_probs, targ_update_num = data_q
333            self.set_targ_update_num(targ_update_num)
334            self.initialize_fns(targ_update_num)
335
336            step_to_pathperf: Dict[int, PathFindPerf] = dict()
337            while True:
338                batch_size = to_q.get()
339                if batch_size is None:
340                    break
341
342                pathfind: P = self.get_pathfind()
343                self._set_pathfind_nnet_fns(pathfind)
344
345                insts_rem_all: List[Instance] = []
346                insts_rem_last_itr: List[Instance] = []
347                for _ in range(self.up_args.search_itrs):
348                    # add instances
349                    self._add_instances(pathfind, insts_rem_last_itr, batch_size, step_probs, times)
350                    assert len(pathfind.instances) == batch_size, f"Values were {len(pathfind.instances)} and {batch_size}"
351
352                    # step
353                    data: List[NDArray] = self._step(pathfind, times)
354                    if self.up_args.sync_main:
355                        _put_from_q([data], from_q, times)
356
357                    # remove instances
358                    insts_rem_last_itr = pathfind.remove_finished_instances(self.up_args.search_itrs)
359                    insts_rem_all.extend(insts_rem_last_itr)
360
361                if not self.up_args.sync_main:
362                    _put_from_q([self._get_instance_data(insts_rem_all + pathfind.instances, times)], from_q, times)
363
364                # pathfinding performance
365                start_time = time.time()
366                self._update_perf(insts_rem_all, step_to_pathperf)
367                times.record_time("update_perf", time.time() - start_time)
368
369                times.add_times(pathfind.times, path=["pathfinding"])
370
371            from_q.put((times, step_to_pathperf))
372            self.clear_nnet_fn_dict()
373        self.to_main_q = None
374        self.from_main_q = None
375        self.nnet_par_info_main = None
376
377    def _add_instances(self, pathfind: P, insts_rem: List[Instance], batch_size: int, step_probs: List[int],
378                       times: Times) -> None:
379        if (len(pathfind.instances) == 0) or (len(insts_rem) > 0):
380            # get steps generate
381            start_time = time.time()
382            steps_gen: List[int]
383            if len(insts_rem) > 0:
384                steps_gen = [int(inst.inst_info[0]) for inst in insts_rem]
385            else:
386                steps_gen = np.random.choice(self.up_args.step_max + 1, size=batch_size,
387                                             p=np.array(step_probs)).tolist()
388            times.record_time("steps_gen", time.time() - start_time)
389
390            # get instance information and kwargs
391            start_time = time.time()
392            inst_infos: List[Tuple[int]] = [(step_gen,) for step_gen in steps_gen]
393            times.record_time("inst_info", time.time() - start_time)
394
395            instances: List[Instance] = self._make_instances(pathfind, steps_gen, inst_infos, times)
396
397            # add instances
398            start_time = time.time()
399            pathfind.add_instances(instances)
400            times.record_time("inst_add", time.time() - start_time)
