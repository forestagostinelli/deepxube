 32    def push_to_open(self, nodes: List[Node], costs: List[float]) -> None:
 33        for node, cost in zip(nodes, costs, strict=True):
 34            heappush(self.open_set, (cost, self.heappush_count, node))
 35            self.heappush_count += 1

 37    def check_closed(self, nodes: List[Node]) -> List[Node]:
 38        nodes_ret: List[Node] = []
 39        for node in nodes:
 40            path_cost_prev: Optional[float] = self.closed_dict.get(node.state)
 41            if (path_cost_prev is None) or (path_cost_prev > node.path_cost):
 42                self.closed_dict[node.state] = node.path_cost
 43                nodes_ret.append(node)
 44        return nodes_ret
 45
 46    def pop_from_open(self) -> List[Node]:
 47        num_to_pop: int = min(self.batch_size, len(self.open_set))
 48
 49        elems_popped: List[OpenSetElem] = []
 50        for _ in range(num_to_pop):
 51            if random.random() < self.eps:
 52                pop_idx: int = random.randrange(0, len(self.open_set))
 53                elems_popped.append(self.open_set.pop(pop_idx))
 54                heapify(self.open_set)
 55            else:
 56                elems_popped.append(heappop(self.open_set))
 57
 58        nodes_popped: List[Node] = [elem_popped[2] for elem_popped in elems_popped]
 59
 60        assert len(elems_popped) > 0
 61
 62        cost_first: float = elems_popped[0][0]
 63        self.lb = max(cost_first, self.lb)
 64
 65        return nodes_popped
 66
 67    def update_ub(self, nodes: List[Node]) -> None:
 68        # keep solved nodes for training
 69        for node in nodes:
 70            assert node.is_solved is not None
 71            if node.is_solved and (self.ub > node.path_cost):
 72                self.goal_node = node
 73                self.ub = node.path_cost
 74
 75    def finished(self) -> bool:
 76        case1: bool = (self.goal_node is not None) and (self.lb >= (self.weight * self.ub))
 77        case2: bool = len(self.open_set) == 0
 78        return case1 or case2

102        instances: List[InstanceBWAS] = [instance for instance in self.instances if not instance.finished()]
103        if len(instances) == 0:

112        # is solved
113        start_time = time.time()

122        self.times.record_time("ub", time.time() - start_time)
123
124        # expand nodes
125        nodes_c_by_inst: List[List[Node]] = self._expand_nodes(instances, nodes_popped_by_inst)
126
127        # check closed
128        start_time = time.time()
129        for inst_idx, instance in enumerate(instances):
130            nodes_c_by_inst[inst_idx] = instance.check_closed(nodes_c_by_inst[inst_idx])
131        self.times.record_time("check", time.time() - start_time)
132
133        # cost
134        start_time = time.time()
135        nodes_c_flat: List[Node] = misc_utils.flatten(nodes_c_by_inst)[0]
136        weights, split_idxs = misc_utils.flatten([[instance.weight] * len(nodes_c)
137                                                  for instance, nodes_c in
138                                                  zip(instances, nodes_c_by_inst, strict=True)])
139        path_costs: List[float] = [node.path_cost for node in nodes_c_flat]
140        heuristics: List[float] = [node.heuristic for node in nodes_c_flat]
141        costs_flat: List[float] = ((np.array(weights) * np.array(path_costs)) + np.array(heuristics)).tolist()
142        costs_by_inst: List[List[float]] = misc_utils.unflatten(costs_flat, split_idxs)
143        self.times.record_time("cost", time.time() - start_time)
144
145        # push to open
146        start_time = time.time()
147        for instance, nodes_c, costs in zip(instances, nodes_c_by_inst, costs_by_inst, strict=True):
148            instance.push_to_open(nodes_c, costs)
149        self.times.record_time("push", time.time() - start_time)
150
151        # verbose
152        if verbose:
153            if len(heuristics) > 0:
154                min_heur = float(np.min(heuristics))
155                min_heur_pc = float(path_costs[np.argmin(heuristics)])
156                max_heur = float(np.max(heuristics))
157                max_heur_pc = float(path_costs[np.argmax(heuristics)])
158                per_has_soln: float = 100.0 * float(np.mean([inst.has_soln() for inst in self.instances]))
159                per_finished: float = 100.0 * float(np.mean([inst.finished() for inst in self.instances]))
160
161                print(f"Itr: %i, Added to OPEN - Min/Max Heur(PathCost): "
162                      f"%.2E(%.2f)/%.2E(%.2f), %%has_soln: {per_has_soln}, "
163                      f"%%finished: {per_finished}" % (self.itr, min_heur, min_heur_pc, max_heur, max_heur_pc))
164
165            print(f"Times - {self.times.get_time_str()}")
166            print("")

168        # update iterations
169        self.itr += 1
170        for instance in instances:
171            instance.itr += 1

173        # return
174        return nodes_popped_flat

110        nodes_popped_by_inst: List[List[Node]] = [instance.pop_from_open() for instance in instances]
111        self.times.record_time("pop", time.time() - start_time)
112
113        # is solved
114        start_time = time.time()
115        nodes_popped_flat: List[Node] = misc_utils.flatten(nodes_popped_by_inst)[0]
116        self.set_is_solved(nodes_popped_flat)
117        self.times.record_time("is_solved", time.time() - start_time)
118
119        # ub
120        start_time = time.time()
121        for instance, nodes_popped in zip(instances, nodes_popped_by_inst, strict=True):
122            instance.update_ub(nodes_popped)
123        self.times.record_time("ub", time.time() - start_time)
124
125        # expand nodes
126        nodes_c_by_inst: List[List[Node]] = self._expand_nodes(instances, nodes_popped_by_inst)
127
128        # check closed
129        start_time = time.time()
130        for inst_idx, instance in enumerate(instances):
131            nodes_c_by_inst[inst_idx] = instance.check_closed(nodes_c_by_inst[inst_idx])
132        self.times.record_time("check", time.time() - start_time)
133
134        # cost
135        start_time = time.time()
136        nodes_c_flat: List[Node] = misc_utils.flatten(nodes_c_by_inst)[0]
137        weights, split_idxs = misc_utils.flatten([[instance.weight] * len(nodes_c)
138                                                  for instance, nodes_c in
139                                                  zip(instances, nodes_c_by_inst, strict=True)])
140        path_costs: List[float] = [node.path_cost for node in nodes_c_flat]
141        heuristics: List[float] = [node.heuristic for node in nodes_c_flat]
142        costs_flat: List[float] = ((np.array(weights) * np.array(path_costs)) + np.array(heuristics)).tolist()
143        costs_by_inst: List[List[float]] = misc_utils.unflatten(costs_flat, split_idxs)
144        self.times.record_time("cost", time.time() - start_time)
145
146        # push to open
147        start_time = time.time()
148        for instance, nodes_c, costs in zip(instances, nodes_c_by_inst, costs_by_inst, strict=True):
149            instance.push_to_open(nodes_c, costs)
150        self.times.record_time("push", time.time() - start_time)
151
152        # verbose
153        if verbose:
154            if len(heuristics) > 0:
155                min_heur = float(np.min(heuristics))
156                min_heur_pc = float(path_costs[np.argmin(heuristics)])
157                max_heur = float(np.max(heuristics))
158                max_heur_pc = float(path_costs[np.argmax(heuristics)])
159                per_has_soln: float = 100.0 * float(np.mean([inst.has_soln() for inst in self.instances]))
160                per_finished: float = 100.0 * float(np.mean([inst.finished() for inst in self.instances]))
161
162                print(f"Itr: %i, Added to OPEN - Min/Max Heur(PathCost): "
163                      f"%.2E(%.2f)/%.2E(%.2f), %%has_soln: {per_has_soln}, "
164                      f"%%finished: {per_finished}" % (self.itr, min_heur, min_heur_pc, max_heur, max_heur_pc))
165
166            print(f"Times - {self.times.get_time_str()}")
167            print("")
168
169        # update iterations
170        self.itr += 1
171        for instance in instances:
172            instance.itr += 1
173
174        # return
175        return nodes_popped_flat
